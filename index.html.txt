  <script>
    // Zodiac order using the common formula: (year - 4) % 12 => Rat is 0
    const ZODIAC_ORDER = [
      "rat","ox","tiger","rabbit","dragon","snake",
      "horse","goat","monkey","rooster","dog","pig"
    ];

    // Quick, commonly-cited "headline" traits (keep yours editable)
    const ZODIAC_INFO = {
      rat:    { label: "Rat",    traits: ["Quick-witted","Resourceful","Adaptable","Persuasive"] },
      ox:     { label: "Ox",     traits: ["Reliable","Patient","Determined","Strong"] },
      tiger:  { label: "Tiger",  traits: ["Brave","Confident","Competitive","Unpredictable"] },
      rabbit: { label: "Rabbit", traits: ["Gentle","Diplomatic","Cautious","Empathetic"] },
      dragon: { label: "Dragon", traits: ["Ambitious","Charismatic","Energetic","Bold"] },
      snake:  { label: "Snake",  traits: ["Wise","Private","Intuitive","Strategic"] },
      horse:  { label: "Horse",  traits: ["Independent","Active","Sociable","Free-spirited"] },
      goat:   { label: "Goat",   traits: ["Creative","Calm","Kind","Idealistic"] },
      monkey: { label: "Monkey", traits: ["Clever","Playful","Curious","Inventive"] },
      rooster:{ label: "Rooster",traits: ["Observant","Hardworking","Direct","Confident"] },
      dog:    { label: "Dog",    traits: ["Loyal","Honest","Protective","Fair-minded"] },
      pig:    { label: "Pig",    traits: ["Generous","Warm","Sincere","Easygoing"] },
    };

    // Build a list of years for a given animal in a range.
    function yearsForAnimal(animalKey, fromYear = 1924, toYear = 2043) {
      const idx = ZODIAC_ORDER.indexOf(animalKey);
      const years = [];
      for (let y = fromYear; y <= toYear; y++) {
        const zodiacIndex = ((y - 4) % 12 + 12) % 12;
        if (zodiacIndex === idx) years.push(y);
      }
      return years;
    }

    // Optional: find the zodiac animal for a specific year (if you also add a "type a year" input later)
    function animalForYear(year) {
      const i = ((year - 4) % 12 + 12) % 12;
      return ZODIAC_ORDER[i];
    }

    // --- UI ---
    const grid = document.getElementById("grid");
    const backdrop = document.getElementById("backdrop");
    const closeBtn = document.getElementById("closeBtn");
    const modalTitle = document.getElementById("modalTitle");
    const traitPills = document.getElementById("traitPills");
    const yearsBox = document.getElementById("yearsBox");
    const yearFromEl = document.getElementById("yearFrom");
    const yearToEl = document.getElementById("yearTo");

    // Map each tile to a file. Rename these to match your exported tiles.
    const TILE_IMAGES = {
      rat: "tiles/rat.png",
      ox: "tiles/ox.png",
      tiger: "tiles/tiger.png",
      rabbit: "tiles/rabbit.png",
      dragon: "tiles/dragon.png",
      snake: "tiles/snake.png",
      horse: "tiles/horse.png",
      goat: "tiles/goat.png",
      monkey: "tiles/monkey.png",
      rooster: "tiles/rooster.png",
      dog: "tiles/dog.png",
      pig: "tiles/pig.png",
    };

    function openModal(animalKey) {
      const info = ZODIAC_INFO[animalKey];
      const fromYear = 1924;
      const toYear = 2043;
      const years = yearsForAnimal(animalKey, fromYear, toYear);

      modalTitle.textContent = `Year of the ${info.label}`;
      traitPills.innerHTML = info.traits.map(t => `<span class="pill">${t}</span>`).join("");
      yearFromEl.textContent = fromYear;
      yearToEl.textContent = toYear;
      yearsBox.textContent = years.join(", ");

      backdrop.classList.add("open");
    }

    function closeModal() {
      backdrop.classList.remove("open");
    }

    // Render grid
    ZODIAC_ORDER.forEach(animalKey => {
      const btn = document.createElement("button");
      btn.className = "tile";
      btn.type = "button";
      btn.setAttribute("data-animal", animalKey);

      const img = document.createElement("img");
      img.src = TILE_IMAGES[animalKey];
      img.alt = ZODIAC_INFO[animalKey].label;

      btn.appendChild(img);
      btn.addEventListener("click", () => openModal(animalKey));
      grid.appendChild(btn);
    });

    closeBtn.addEventListener("click", closeModal);
    backdrop.addEventListener("click", (e) => {
      if (e.target === backdrop) closeModal();
    });
    window.addEventListener("keydown", (e) => {
      if (e.key === "Escape") closeModal();
    });
  </script>
